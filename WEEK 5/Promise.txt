
const myPromise = new Promise((resolve, reject) => {
    
    let success = true; // Change to false to test rejection
    
    
    if (success) {
        resolve("Operation Successful!");
    } else {
        reject("Operation Failed!");
    }
});


myPromise.then((message) => {
    console.log(message);
})

.catch((error) => {
    console.log(error);
});

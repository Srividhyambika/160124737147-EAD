
function fetchData() {
    return new Promise((resolve, reject) => {
        
       
        setTimeout(() => {
            
           
            resolve("Data fetched successfully!");
            
        }, 2000);
    });
}


async function showData() {
    console.log("Fetching data...");
    
    
    let result = await fetchData();
    
    
    console.log(result);
}


showData();
